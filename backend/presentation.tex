\begin{frame}{Problem and Motivation}
  \begin{itemize}
    \item Traditional map quiz apps are static and repetitive.
    \item Players of different skill levels need different challenge intensity.
    \item Admin moderation is needed for user-generated locations.
  \end{itemize}
\end{frame}

\begin{frame}{Project Goals}
  \begin{itemize}
    \item Build a game-style Serbia geography experience.
    \item Implement secure auth and role-based admin features.
    \item Add adaptive difficulty that reacts to player performance.
    \item Provide measurable outcomes with tests and analytics.
  \end{itemize}
\end{frame}

\begin{frame}{Architecture}
  \begin{itemize}
    \item Backend: FastAPI + Tortoise ORM + PostgreSQL
    \item Frontend: Vue 3 + Vue Router + Axios + Leaflet
    \item Auth: HTTP-only cookie JWT sessions
    \item Domains: users, locations, games, admin
  \end{itemize}
\end{frame}

\begin{frame}{Core Data Model}
  \begin{itemize}
    \item \texttt{users}, \texttt{locations}, \texttt{game\_sessions}, \texttt{rounds}
    \item Adaptive tables:
    \begin{itemize}
      \item \texttt{user\_skill\_profile}
      \item \texttt{location\_difficulty\_profile}
      \item \texttt{adaptive\_decision\_log}
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Adaptive Difficulty Logic}
  \begin{itemize}
    \item User skill computed from recent points and distances.
    \item Location difficulty computed from historical outcomes.
    \item Challenge start:
    \begin{itemize}
      \item \texttt{mode=adaptive}: pick rounds near user skill
      \item \texttt{mode=fixed}: random selection
    \end{itemize}
    \item Every round updates both user and location profiles.
  \end{itemize}
\end{frame}

\begin{frame}{Main API Endpoints}
  \begin{itemize}
    \item \texttt{POST /api/v1/game/start}
    \item \texttt{POST /api/v1/game/play}
    \item \texttt{GET /api/v1/profile/adaptive-stats}
    \item \texttt{GET /api/v1/admin/adaptive/stats}
    \item \texttt{GET /api/v1/game/leaderboard/daily}
    \item \texttt{GET /api/v1/game/leaderboard/monthly}
  \end{itemize}
\end{frame}

\begin{frame}{Admin Workflow}
  \begin{itemize}
    \item Users can submit new locations.
    \item Non-admin submissions are pending by default.
    \item Admin can approve/reject pending locations.
    \item Admin-created locations are auto-approved in service layer.
  \end{itemize}
\end{frame}
